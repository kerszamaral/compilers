
int ai = 51;
int bi = 1;

int di = 4;

char ac = 51;
char bc = 1;
char cc = 2;
char dc = 4;

int af = 51/1;

int cf = 2/1;

int main()
{
    print "Initial int: " ai "\n";
    ai = ai + bi;
    print "iAdd: " ai "\n";
    ai = ai - bi;
    print "iSub: " ai "\n";
    ai = ai * ci;
    print "iMul: " ai "\n";
    ai = ai / ci;
    print "iDiv: " ai "\n";
    ai = ai % di;
    print "iMod: " ai "\n";

    print "Initial char: " ac "\n";
    ac = ac + bc;
    print "cAdd: " ac "\n";
    ac = ac - bc;
    print "cSub: " ac "\n";
    ac = ac * cc;
    print "cMul: " ac "\n";
    ac = ac / cc;
    print "cDiv: " ac "\n";
    ac = ac % dc;
    print "cMod: " ac "\n";

    print "Initial Real: " af "\n";
    af = af + bf;
    print "fAdd: " af "\n";
    af = af - bf;
    print "fSub: " af "\n";
    af = af * cf;
    print "fMul: " af "\n";
    af = af / cf;
}
